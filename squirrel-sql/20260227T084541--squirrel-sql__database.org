(let ((jdbc-dir (expand-file-name "_jdbc")))
  (mkdir jdbc-dir t)

  (let ((default-directory jdbc-dir)
        (main-domain (auth-source-pick-first-password :host "mydomain" :user "main"))
        (curl-common-args '("-fsSL"))
        (pgsql-version "42.7.10")
        (sqlite-version "3.51.2.0")
        (mysql-version "9.6.0")
        (mod-date "date.txt"))

    (if (file-exists-p mod-date)
        (progn
          (delete-file mod-date)
          (write-region "" nil mod-date))
      (write-region "" nil mod-date)
      (setq mod-date "Jan 01, 1982"))
    
    (apply #'call-process "curl" nil 0 nil
           "-z" mod-date
           (format "https://xget.%s/gh/xerial/sqlite-jdbc/releases/download/%s/sqlite-jdbc-%s.jar"
                   main-domain sqlite-version sqlite-version)
           "-O" curl-common-args)

    (apply #'call-process "curl" nil 0 nil
           "--etag-compare" "pgsql.txt" "--etag-save" "pgsql.txt"
           (format "https://jdbc.postgresql.org/download/postgresql-%s.jar" pgsql-version)
           "-O" curl-common-args)

    (call-process-shell-command (format "curl %s -z \"%s\" https://dev.mysql.com/get/Downloads/Connector-J/mysql-connector-j-%s.tar.gz | tar -xzf - --include */mysql-connector-j-%s.jar --strip-components 1" (mapconcat #'shell-quote-argument curl-common-args " ") mod-date mysql-version mysql-version) nil 0)))
