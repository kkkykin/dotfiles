       (results (directory-files result-dir t "^[^.]+\\.csv$"))
       (result "104.17.133.211"))
  (when results
    (let ((latest (car (last results))))
      (with-temp-buffer
        (insert-file-contents latest)
        (goto-char (point-min))
        (forward-line)
        (let ((pos (point)))
