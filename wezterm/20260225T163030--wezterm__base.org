-- 返回 true 表示端口已被占用（不可绑定）
local function port_in_use(port)
  if is_windows then
    -- Windows: 用 PowerShell 查询监听端口
    local ok, _, _ = wezterm.run_child_process({
      'powershell', '-NoProfile', '-Command',
      string.format(
        "if (Get-NetTCPConnection -State Listen -LocalPort %d -ErrorAction SilentlyContinue) { exit 0 } else { exit 1 }",
        port
      ),
    })
    return ok
  else
    -- macOS/Linux: 用 python3 尝试 bind 127.0.0.1:port
    -- 约定：exit 0 = in use；exit 1 = free
    local script = string.format([[
python3 - <<'PY'
import socket, sys
port = %d
s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
try:
    s.bind(("127.0.0.1", port))
except OSError:
    sys.exit(0)  # in use
else:
    sys.exit(1)  # free
finally:
    s.close()
PY
]], port)

    local ok, _, _ = wezterm.run_child_process({ 'sh', '-lc', script })
    return ok
  end
end

-- 主题缓存和工具函数
local theme_utils = {}
local tab_theme = {}

local favorite_themes = {
  'Bluloco Zsh Light (Gogh)',   -- crawl
  'Ef-Tritanopia-Light',
  'Atelier Savanna Light (base16)',
}

-- https://wezterm.org/config/lua/wezterm.gui/get_appearance.html
-- wezterm.gui is not available to the mux server, so take care to
-- do something reasonable when this config is evaluated by the mux
function theme_utils.get_appearance()
  if wezterm.gui then
    return wezterm.gui.get_appearance()
  end
  return 'dark'
end

-- 计算主题亮度
function theme_utils.get_brightness(scheme)
  local bg = scheme.background or '#000000'
  if not bg:find('^#%x%x%x%x%x%x$') then
    return 0
  end

  local r = tonumber(bg:sub(2, 3), 16)
  local g = tonumber(bg:sub(4, 5), 16)
  local b = tonumber(bg:sub(6, 7), 16)
  return (r + g + b) / 3
end

-- 获取指定类型的主题（带缓存）
function theme_utils.get_themes_by_type(theme_type)
  local cache_key = 'theme_cache_' .. theme_type

  -- 检查缓存
  if wezterm.GLOBAL[cache_key] then
    return wezterm.GLOBAL[cache_key]
  end

  -- 计算并缓存
  local schemes = wezterm.get_builtin_color_schemes()
  local themes = {}

  for name, scheme in pairs(schemes) do
    local brightness = theme_utils.get_brightness(scheme)
    local is_dark = brightness <= 128

    if (theme_type == 'dark' and is_dark) or
       (theme_type == 'light' and not is_dark) then
      table.insert(themes, name)
    end
  end

  -- 创建查找表，快速获取主题在喜欢列表中的位置
  local favorite_index = {}
  for i, theme in ipairs(favorite_themes) do
    favorite_index[theme] = i
  end

  -- 自定义排序：按 favorite_themes 的顺序，然后按字母顺序
  table.sort(themes, function(a, b)
    local a_idx = favorite_index[a]
    local b_idx = favorite_index[b]

    -- 如果都是喜欢的主题，按在 favorite_themes 中的顺序排列
    if a_idx and b_idx then
      return a_idx < b_idx
    end

    -- 如果 a 是喜欢的主题而 b 不是，a 排在前面
    if a_idx and not b_idx then
      return true
    end

    -- 如果 b 是喜欢的主题而 a 不是，b 排在前面
    if not a_idx and b_idx then
      return false
    end

    -- 如果都不是喜欢的主题，按字母顺序排列
    return a < b
  end)

  wezterm.GLOBAL[cache_key] = themes
  return themes
end

function theme_utils.set_theme(window, theme)
  tab_theme[window:active_tab():tab_id()] = theme
  window:set_config_overrides { color_scheme = theme }
  wezterm.log_info('Applied theme: ' .. theme)
end

-- 创建主题选择器
function theme_utils.create_theme_selector(theme_type)
  local themes = theme_utils.get_themes_by_type(theme_type)
  local items = {}

  for _, name in ipairs(themes) do
    table.insert(items, { id = name, label = name })
  end

  return act.InputSelector {
    title = 'Select ' .. (theme_type == 'dark' and 'Dark' or 'Light') .. ' Theme',
    choices = items,
    fuzzy = true,
    action = wezterm.action_callback(function(window, pane, id, label)
      if id then
        theme_utils.set_theme(window, id)
      end
    end),
  }
end

function theme_utils.set_random_theme_matching_appearance(window, pane)
  local appearance = theme_utils.get_appearance()
  local theme_type = appearance:find('Dark') and 'dark' or 'light'

  local themes = theme_utils.get_themes_by_type(theme_type)
  if #themes > 0 then
    local random_theme = themes[math.random(#themes)]
    theme_utils.set_theme(window, random_theme)
  end
end

local emacs_utils = {}

function emacs_utils.escape_elisp_string(s)
  return (s:gsub("\\", "\\\\"):gsub('"', '\\"'))
end

function emacs_utils.eval(sexp, cb)
  cb = cb or function(_) end

  local success, stdout, stderr = wezterm.run_child_process({
    "emacsclient", "-n", "-e", sexp
  })

  cb({
    success = success,
    stdout = stdout,
    stderr = stderr
  })
end

function emacs_utils.search_dictionary(word)
  local sexp = string.format('(dictionary-search "%s")',
                            emacs_utils.escape_elisp_string(word))
  emacs_utils.eval(sexp, function(res)
    if not res.success then
      local msg = (res.stderr and res.stderr ~= "" and res.stderr)
        or ("emacsclient exited with error")
      wezterm.log_error(msg)
    end
  end)
end

  -- local tab, pane, window = mux.spawn_window(cmd or {})

  local tab, pane, window = mux.spawn_window({
    args = { 'nvim', '--listen', default_nvim_server, '+te powershell', '+star' },
    cwd = os.getenv("USERPROFILE") or os.getenv("HOME")
  })
  if tab then
    tab:set_title('nvim')
  end

  if is_windows then
    local tab, pane, window = window:spawn_tab({
      args = { 'cmd', '/k', init_windows_script, 'init' },
    })
  end
end)

wezterm.on('update-status', function(window, pane)
  local active_tab = window:active_tab()
  if active_tab then
    local saved_theme = tab_theme[active_tab:tab_id()]
    local current_theme = window:effective_config().color_scheme

    if saved_theme and current_theme and saved_theme ~= current_theme then
      theme_utils.set_theme(window, saved_theme)
    end
  end
end)
